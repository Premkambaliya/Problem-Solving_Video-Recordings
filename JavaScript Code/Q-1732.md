# Find the Highest Altitude (JavaScript)

```javascript

/**
 * @param {number[]} gain
 * @return {number}
 */
var largestAltitude = function(gain) {
    let maxAltitudes=0;
    let current=0;

    for(let i=0;i<gain.length;i++){
        current+=gain[i];
        maxAltitudes=Math.max(maxAltitudes, current);
    }
    return maxAltitudes;
};
