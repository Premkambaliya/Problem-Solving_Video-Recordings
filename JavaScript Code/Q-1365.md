# How Many Numbers Are Smaller Than the Current Number (JavaScript)

```javascript

/**
 * @param {number[]} nums
 * @return {number[]}
 */
var smallerNumbersThanCurrent = function(nums) {
    let result = [];

    for(let i=0;i<nums.length;i++){
        let count = 0;
        for(let j=0;j<nums.length;j++){
            if( j !== i && nums[j]<nums[i]){
                count++;
            }
        }
        result.push(count);
    }

    return result;
};
