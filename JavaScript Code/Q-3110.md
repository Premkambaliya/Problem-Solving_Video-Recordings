# Score of a String (JavaScript)

```javascript

/**
 * @param {string} s
 * @return {number}
 */
var scoreOfString = function(s) {
    let temp = 0;
    for(let i=0;i<s.length-1;i++){
        let diff = s.charCodeAt(i) - s.charCodeAt(i+1);
        if(diff<0){
            diff = diff - (diff*2);
        }
        temp = temp + diff;
    }
    return temp;
};
