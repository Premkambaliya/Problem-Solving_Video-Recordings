# Richest Customer wealth (JavaScript)

```javascript

/**
 * @param {number[][]} accounts
 * @return {number}
 */
var maximumWealth = function(accounts) {
    let maxwealth = 0;

    for(let i=0;i<accounts.length;i++){
        let sum = accounts[i].reduce((acc,curr)=>acc+curr,0);

        if(sum>maxwealth){
           maxwealth=sum;
        }
    }
    return maxwealth;
};
