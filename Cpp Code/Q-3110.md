# Score of a String (C++)

```cpp

class Solution {
public:
    int scoreOfString(string s) {
        int temp = 0;
        for(int i=0;i<s.length()-1;i++){
            int diff = s[i] - s[i+1];

            if(diff<0){
                diff = diff - (diff*2);
            }
            temp = temp + diff;
        }
        return temp;
    }
};
