# Max Consecutive Ones (C++)

```cpp

class Solution {
public:
    int findMaxConsecutiveOnes(vector<int>& nums) {
        int maxcount=0;
        int current=0;

        for(int i=0;i<nums.size();i++){
            int num=nums[i];

            if(num==1){
                current++;
                maxcount= max(maxcount, current);
            }else{
                current=0;
            }
        }
        return maxcount;
    }
};
