1877. Minimize Maximum Pair Sum in Array

class Solution {
public:
    int minPairSum(vector<int>& nums) {
        sort(nums.begin(), nums.end());
        int n = nums.size();
        int sum = 0;
        for(int i=0;i<n;i++){sum = max(sum, nums[i] + nums[n-1-i]);
        }
        return sum;
    }
};
